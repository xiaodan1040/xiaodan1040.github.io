# 用于修复被装饰对象的名称空间
from functools import wrape

def wapper(func):
    @wraps(func)
    def inner(*args, **kwargs):
        # func执行前的操作
        res = func(*args, **kwargs)
        # func执行后的操作
        return res
    return inner

2、你所知道的能够实现单例模式的方式有哪些，尝试写几个

单例模式：让所有类在实例化时，指向同一内存地址

实现单例模式的5种方式：

  • 通过关键字classmethod 实现
  • 通过装饰器实现
  • 通过__new__ 魔法方法实现
  • 通过导入模块实现
  • 通过元类实现
# 通过关键字classmethod 实现
class Singleton:
    def __init__(self, *args, **kwargs):
        pass
    
    @classmethod
    def instance(cls, *args, **kwargs):
        if not hasattr(cls, '_instance'):
            cls._instance = cls(*args, **kwargs)
        return cls._instance

# 通过装饰器实现
def singleton(cls):
    _instance = {}
    
    def inner(*args, **kwargs):
        if cls not in _instance:
            _instance[cls] = cls(*args, **kwargs)
        return _instance[cls]
    return inner

# 通过__new__ 魔法方法实现
class Singleton:
    def __new__(cls, *args, **kwargs):
        if not hasattr(cls, '_instance'):
            cls._instance = super().__new__(cls)
        return cls._instance

# 通过导入模块实现
# a.py 被导入的模块
class Singleton:
    def __init__(self):
        pass
    
    singleton_obj = Singleton()

# b.py 导入a.py 模块
from a import singleton_obj
print singleton_obj

# 通过元类实现
class SingletonMeta(type):
    def __call__(cls, *args, **kwargs):
        if not hasattr(cls, '_instance'):
            cls._instance = super().__call__(*args, **kwargs)
        return cls._instance

class Singleton(metaclass=SingletonMeta):
    def __init__(self):
        pass

3、举例说明map、zip、filter、sorted、reduce 内置函数的作用

*map(func, iterable)

映射：将函数func用于可迭代对象中的元素，所得到的每个返回值映射到一个map 对象中，可将这个map 对象转换其他容器类型展示结果。只能转换一次，原理：map对象本质是可迭代对象。tuple(map_obj) 或for num in map_obj这样的语句，就是调用了迭代器，执行了 __next__()，再次使用map_obj，里面的内容已被取完

'''求列表[1,2,3,4,5,6,7,8,9],返回一个n*n 的列表
'''
li = [1, 2, 3, 4, 5, 6, 7, 8, 9]
map_obj = map(lambda x: x ** 2, li)
print(f'map 对象：{map_obj}')   >>> map 对象：<map object at 0x000001F6AEA62588>
print(f'转换成元组展示：{tuple(map_obj)}')   >>> 转换成元组展示：(1, 4, 9, 16, 25, 36, 49, 64, 81)
print(f'转换成列表展示：{list(map_obj)}')   >>> 转换成列表展示：[]

*zip(iterable)

并行迭代：将多个可迭代对象中元素，一一对应打包成一个个元组，将这些元组存放在zip对象中。参数序列不等长，会自动甄别，忽略多余的元素

li1 = ['name', 'age', 'hobby']
li2 = ['dawn', 28]
zip_obj = zip(li1, li2)
print(zip_obj)   >>> <zip object at 0x0000023ECA4C87C8>
print(dict(zip_obj))   >>> {'name': 'dawn', 'age': 28}
print(tuple(zip_obj))   >>> ()

filter(func, iterable)

过滤：通过函数fun判断可迭代对象中的元素是否符合条件，筛选符合条件的元素存放到filter 对象中

'''求列表['1A','2A','3C','4C','5A']中，返回不包含A的列表
'''
key_list = ['1A', '2A', '3C', '4C', '5A']
filter_obj = filter(lambda key: key.endswith('C'), key_list)

print(f'filter 对象：{filter_obj}')   >>> filter 对象：<filter object at 0x000002D05C3A95C8>
print(f'转换成元组展示：{tuple(filter_obj)}')   >>> 转换成元组展示：('3C', '4C')
print(f'转换成列表展示：{list(filter_obj)}')   >>> 转换成列表展示：[]，原理同map()

sorted(iterable, func, reverser=False)

排序：通过函数func制定排序规则，对可迭代对象中的元素进行排序，默认为升序，返回一个新的列表

tuple_obj = ('ada', 'afdf', 'asdasdsasda', 'asdasdf', 'sss')
list_obj = sorted(tuple_obj, key=lambda s: s.count('a'), reverse=True)
print(list_obj) >>> ['asdasdsasda', 'ada', 'asdasdf', 'afdf', 'sss']
print(tuple(list_obj)) >>> ('asdasdsasda', 'ada', 'asdasdf', 'afdf', 'sss')

reduce(func, sequence, initial=None)

合并：通过函数对参数序列中元素进行累积计算，即：reduce(func,[1,2,3]) <=> func(func(1,2),3)。注意：使用reduce，需要导入functools 模块，且求和累积，初始值为0，乘积累积，初始值为1

from functools import reduce

# 求1-100之内的和
res_sum = reduce(lambda x, y: x + y, range(1, 101))
print(res_sum)

# 求1-9的积
res_pro = reduce(lambda x, y: x * y, range(1, 10))
print(res_pro)

5、什么是元类?元类的应用场景有哪些，能具体说说吗？

在python 中，万物皆对象。对象是通过类实例化的结果，类是元类实例化的结果，也就是创建类的类。

一个类在实例化的过程中，做了以下事情

  1. 触发元类(创建这个类的类)的 __call__ 方法（通过对象+()的方式触发）
  2. 通过 __new__ 方法产生一个空对象（在 __init__触发前触发）
  3. 通过 __init__方法初始化这个对象（在实例化的时候触发）
  4. 返回这个对象

元类的应用场景

  1. ORM(Object Relational Mapping) , 即对象-关系映射,就是将关系数据库的一行记录映射为一个对象,一个类对应一个表
